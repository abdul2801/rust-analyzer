29pub(crate) fn complete_postfix(
30    acc: &mut Completions,
31    ctx: &CompletionContext<'_>,
32    dot_access: &DotAccess<'_>,
33) {
34    if !ctx.config.enable_postfix_completions {
35        return;
36    }
37
38    let (dot_receiver, receiver_ty, receiver_is_ambiguous_float_literal) = match dot_access {
39        DotAccess { receiver_ty: Some(ty), receiver: Some(it), kind, .. } => (
40            it,
41            &ty.original,
42            match *kind {
43                DotAccessKind::Field { receiver_is_ambiguous_float_literal } => {
44                    receiver_is_ambiguous_float_literal
45                }
46                DotAccessKind::Method => false,
47            },
48        ),
49        _ => return,
50    };
51    let expr_ctx = &dot_access.ctx;
52
53    let receiver_text =
54        get_receiver_text(&ctx.sema, dot_receiver, receiver_is_ambiguous_float_literal);
55
56    let cap = match ctx.config.snippet_cap {
57        Some(it) => it,
58        None => return,
59    };
60
61    let postfix_snippet = match build_postfix_snippet_builder(ctx, cap, dot_receiver) {
62        Some(it) => it,
63        None => return,
64    };
65
66    let cfg = ctx.config.find_path_config(ctx.is_nightly);
67
68    if let Some(drop_trait) = ctx.famous_defs().core_ops_Drop()
69        && receiver_ty.impls_trait(ctx.db, drop_trait, &[])
70        && let Some(drop_fn) = ctx.famous_defs().core_mem_drop()
71        && let Some(path) = ctx.module.find_path(ctx.db, ItemInNs::Values(drop_fn.into()), cfg)
72    {
73        cov_mark::hit!(postfix_drop_completion);
74        let mut item = postfix_snippet(
75            "drop",
76            "fn drop(&mut self)",
77            &format!("{path}($0{receiver_text})", path = path.display(ctx.db, ctx.edition)),
78        );
79        item.set_documentation(drop_fn.docs(ctx.db));
80        item.add_to(acc, ctx.db);
81    }
82
83    postfix_snippet("ref", "&expr", &format!("&{receiver_text}")).add_to(acc, ctx.db);
84    postfix_snippet("refm", "&mut expr", &format!("&mut {receiver_text}")).add_to(acc, ctx.db);
85    postfix_snippet("deref", "*expr", &format!("*{receiver_text}")).add_to(acc, ctx.db);
86
87    // The rest of the postfix completions create an expression that moves an argument,
88    // so it's better to consider references now to avoid breaking the compilation
89
90    let (dot_receiver_including_refs, prefix) = include_references(dot_receiver);
91    let mut receiver_text =
92        get_receiver_text(&ctx.sema, dot_receiver, receiver_is_ambiguous_float_literal);
93    receiver_text.insert_str(0, &prefix);
94    let postfix_snippet =
95        match build_postfix_snippet_builder(ctx, cap, &dot_receiver_including_refs) {
96            Some(it) => it,
97            None => return,
98        };
99
100    if !ctx.config.snippets.is_empty() {
101        add_custom_postfix_completions(acc, ctx, &postfix_snippet, &receiver_text);
102    }
103
104    postfix_snippet("box", "Box::new(expr)", &format!("Box::new({receiver_text})"))
105        .add_to(acc, ctx.db);
106    postfix_snippet("dbg", "dbg!(expr)", &format!("dbg!({receiver_text})")).add_to(acc, ctx.db); // fixme
107    postfix_snippet("dbgr", "dbg!(&expr)", &format!("dbg!(&{receiver_text})")).add_to(acc, ctx.db);
108    postfix_snippet("call", "function(expr)", &format!("${{1}}({receiver_text})"))
109        .add_to(acc, ctx.db);
110
111    let try_enum = TryEnum::from_ty(&ctx.sema, &receiver_ty.strip_references());
112    let mut is_in_cond = false;
113    if let Some(parent) = dot_receiver_including_refs.syntax().parent()
114        && let Some(second_ancestor) = parent.parent()
115    {
116        if let Some(parent_expr) = ast::Expr::cast(parent) {
117            is_in_cond = is_in_condition(&parent_expr);
118        }
119        match &try_enum {
120            Some(try_enum) if is_in_cond => match try_enum {
121                TryEnum::Result => {
122                    postfix_snippet("let", "let Ok(_)", &format!("let Ok($0) = {receiver_text}"))
123                        .add_to(acc, ctx.db);
124                    postfix_snippet(
125                        "letm",
126                        "let Ok(mut _)",
127                        &format!("let Ok(mut $0) = {receiver_text}"),
128                    )
129                    .add_to(acc, ctx.db);
130                }
131                TryEnum::Option => {
132                    postfix_snippet(
133                        "let",
134                        "let Some(_)",
135                        &format!("let Some($0) = {receiver_text}"),
136                    )
137                    .add_to(acc, ctx.db);
138                    postfix_snippet(
139                        "letm",
140                        "let Some(mut _)",
141                        &format!("let Some(mut $0) = {receiver_text}"),
142                    )
143                    .add_to(acc, ctx.db);
144                }
145            },
146            _ if matches!(second_ancestor.kind(), STMT_LIST | EXPR_STMT) => {
147                postfix_snippet("let", "let", &format!("let $0 = {receiver_text};"))
148                    .add_to(acc, ctx.db);
149                postfix_snippet("letm", "let mut", &format!("let mut $0 = {receiver_text};"))
150                    .add_to(acc, ctx.db);
151            }
152            _ => (),
153        }
154    }
155
156    if !is_in_cond {
157        match try_enum {
158            Some(try_enum) => match try_enum {
159                TryEnum::Result => {
160                    postfix_snippet(
161                    "match",
162                    "match expr {}",
163                    &format!("match {receiver_text} {{\n    Ok(${{1:_}}) => {{$2}},\n    Err(${{3:_}}) => {{$0}},\n}}"),
164                )
165                .add_to(acc, ctx.db);
166                }
167                TryEnum::Option => {
168                    postfix_snippet(
169                    "match",
170                    "match expr {}",
171                    &format!(
172                        "match {receiver_text} {{\n    Some(${{1:_}}) => {{$2}},\n    None => {{$0}},\n}}"
173                    ),
174                )
175                .add_to(acc, ctx.db);
176                }
177            },
178            None => {
179                postfix_snippet(
180                    "match",
181                    "match expr {}",
182                    &format!("match {receiver_text} {{\n    ${{1:_}} => {{$0}},\n}}"),
183                )
184                .add_to(acc, ctx.db);
185            }
186        }
187        if let Some(try_enum) = &try_enum {
188            match try_enum {
189                TryEnum::Result => {
190                    postfix_snippet(
191                        "ifl",
192                        "if let Ok {}",
193                        &format!("if let Ok($1) = {receiver_text} {{\n    $0\n}}"),
194                    )
195                    .add_to(acc, ctx.db);
196
197                    postfix_snippet(
198                        "lete",
199                        "let Ok else {}",
200                        &format!("let Ok($1) = {receiver_text} else {{\n    $2\n}};\n$0"),
201                    )
202                    .add_to(acc, ctx.db);
203
204                    postfix_snippet(
205                        "while",
206                        "while let Ok {}",
207                        &format!("while let Ok($1) = {receiver_text} {{\n    $0\n}}"),
208                    )
209                    .add_to(acc, ctx.db);
210                }
211                TryEnum::Option => {
212                    postfix_snippet(
213                        "ifl",
214                        "if let Some {}",
215                        &format!("if let Some($1) = {receiver_text} {{\n    $0\n}}"),
216                    )
217                    .add_to(acc, ctx.db);
218
219                    postfix_snippet(
220                        "lete",
221                        "let Some else {}",
222                        &format!("let Some($1) = {receiver_text} else {{\n    $2\n}};\n$0"),
223                    )
224                    .add_to(acc, ctx.db);
225
226                    postfix_snippet(
227                        "while",
228                        "while let Some {}",
229                        &format!("while let Some($1) = {receiver_text} {{\n    $0\n}}"),
230                    )
231                    .add_to(acc, ctx.db);
232                }
233            }
234        } else if receiver_ty.is_bool() || receiver_ty.is_unknown() {
235            postfix_snippet("if", "if expr {}", &format!("if {receiver_text} {{\n    $0\n}}"))
236                .add_to(acc, ctx.db);
237            postfix_snippet(
238                "while",
239                "while expr {}",
240                &format!("while {receiver_text} {{\n    $0\n}}"),
241            )
242            .add_to(acc, ctx.db);
243            postfix_snippet("not", "!expr", &format!("!{receiver_text}")).add_to(acc, ctx.db);
244        } else if let Some(trait_) = ctx.famous_defs().core_iter_IntoIterator()
245            && receiver_ty.impls_trait(ctx.db, trait_, &[])
246        {
247            postfix_snippet(
248                "for",
249                "for ele in expr {}",
250                &format!("for ele in {receiver_text} {{\n    $0\n}}"),
251            )
252            .add_to(acc, ctx.db);
253        }
254    }
255
256    let mut block_should_be_wrapped = true;
257    if dot_receiver.syntax().kind() == BLOCK_EXPR {
258        block_should_be_wrapped = false;
259        if let Some(parent) = dot_receiver.syntax().parent()
260            && matches!(parent.kind(), IF_EXPR | WHILE_EXPR | LOOP_EXPR | FOR_EXPR)
261        {
262            block_should_be_wrapped = true;
263        }
264    };
265    {
266        let (open_brace, close_brace) =
267            if block_should_be_wrapped { ("{ ", " }") } else { ("", "") };
268        let (open_paren, close_paren) = if is_in_cond { ("(", ")") } else { ("", "") };
269        let unsafe_completion_string =
270            format!("{open_paren}unsafe {open_brace}{receiver_text}{close_brace}{close_paren}");
271        postfix_snippet("unsafe", "unsafe {}", &unsafe_completion_string).add_to(acc, ctx.db);
272
273        let const_completion_string =
274            format!("{open_paren}const {open_brace}{receiver_text}{close_brace}{close_paren}");
275        postfix_snippet("const", "const {}", &const_completion_string).add_to(acc, ctx.db);
276    }
277
278    if let ast::Expr::Literal(literal) = dot_receiver_including_refs.clone()
279        && let Some(literal_text) = ast::String::cast(literal.token())
280    {
281        add_format_like_completions(acc, ctx, &dot_receiver_including_refs, cap, &literal_text);
282    }
283
284    postfix_snippet(
285        "return",
286        "return expr",
287        &format!(
288            "return {receiver_text}{semi}",
289            semi = if expr_ctx.in_block_expr { ";" } else { "" }
290        ),
291    )
292    .add_to(acc, ctx.db);
293
294    if let Some(BreakableKind::Block | BreakableKind::Loop) = expr_ctx.in_breakable {
295        postfix_snippet(
296            "break",
297            "break expr",
298            &format!(
299                "break {receiver_text}{semi}",
300                semi = if expr_ctx.in_block_expr { ";" } else { "" }
301            ),
302        )
303        .add_to(acc, ctx.db);
304    }
305}

307fn get_receiver_text(
308    sema: &Semantics<'_, RootDatabase>,
309    receiver: &ast::Expr,
310    receiver_is_ambiguous_float_literal: bool,
311) -> String {
312    // Do not just call `receiver.to_string()`, as that will mess up whitespaces inside macros.
313    let Some(mut range) = sema.original_range_opt(receiver.syntax()) else {
314        return receiver.to_string();
315    };
316    if receiver_is_ambiguous_float_literal {
317        range.range = TextRange::at(range.range.start(), range.range.len() - TextSize::of('.'))
318    }
319    let file_text = sema.db.file_text(range.file_id.file_id(sema.db));
320    let mut text = file_text.text(sema.db)[range.range].to_owned();
321
322    // The receiver texts should be interpreted as-is, as they are expected to be
323    // normal Rust expressions.
324    escape_snippet_bits(&mut text);
325    text
326}

328/// Escapes `\` and `$` so that they don't get interpreted as snippet-specific constructs.
329///
330/// Note that we don't need to escape the other characters that can be escaped,
331/// because they wouldn't be treated as snippet-specific constructs without '$'.
332fn escape_snippet_bits(text: &mut String) {
333    stdx::replace(text, '\\', "\\\\");
334    stdx::replace(text, '$', "\\$");
335}

337fn include_references(initial_element: &ast::Expr) -> (ast::Expr, String) {
338    let mut resulting_element = initial_element.clone();
339
340    while let Some(field_expr) = resulting_element.syntax().parent().and_then(ast::FieldExpr::cast)
341    {
342        resulting_element = ast::Expr::from(field_expr);
343    }
344
345    let mut prefix = String::new();
346
347    let mut found_ref_or_deref = false;
348
349    while let Some(parent_deref_element) =
350        resulting_element.syntax().parent().and_then(ast::PrefixExpr::cast)
351    {
352        if parent_deref_element.op_kind() != Some(ast::UnaryOp::Deref) {
353            break;
354        }
355
356        found_ref_or_deref = true;
357        resulting_element = ast::Expr::from(parent_deref_element);
358
359        prefix.insert(0, '*');
360    }
361
362    while let Some(parent_ref_element) =
363        resulting_element.syntax().parent().and_then(ast::RefExpr::cast)
364    {
365        found_ref_or_deref = true;
366        let exclusive = parent_ref_element.mut_token().is_some();
367        resulting_element = ast::Expr::from(parent_ref_element);
368
369        prefix.insert_str(0, if exclusive { "&mut " } else { "&" });
370    }
371
372    if !found_ref_or_deref {
373        // If we do not find any ref/deref expressions, restore
374        // all the progress of tree climbing
375        prefix.clear();
376        resulting_element = initial_element.clone();
377    }
378
379    (resulting_element, prefix)
380}

382fn build_postfix_snippet_builder<'ctx>(
383    ctx: &'ctx CompletionContext<'_>,
384    cap: SnippetCap,
385    receiver: &'ctx ast::Expr,
386) -> Option<impl Fn(&str, &str, &str) -> Builder + 'ctx> {
387    let receiver_range = ctx.sema.original_range_opt(receiver.syntax())?.range;
388    if ctx.source_range().end() < receiver_range.start() {
389        // This shouldn't happen, yet it does. I assume this might be due to an incorrect token
390        // mapping.
391        never!();
392        return None;
393    }
394    let delete_range = TextRange::new(receiver_range.start(), ctx.source_range().end());
395
396    // Wrapping impl Fn in an option ruins lifetime inference for the parameters in a way that
397    // can't be annotated for the closure, hence fix it by constructing it without the Option first
398    fn build<'ctx>(
399        ctx: &'ctx CompletionContext<'_>,
400        cap: SnippetCap,
401        delete_range: TextRange,
402    ) -> impl Fn(&str, &str, &str) -> Builder + 'ctx {
403        move |label, detail, snippet| {
404            let edit = TextEdit::replace(delete_range, snippet.to_owned());
405            let mut item = CompletionItem::new(
406                CompletionItemKind::Snippet,
407                ctx.source_range(),
408                label,
409                ctx.edition,
410            );
411            item.detail(detail).snippet_edit(cap, edit);
412            let postfix_match = if ctx.original_token.text() == label {
413                cov_mark::hit!(postfix_exact_match_is_high_priority);
414                Some(CompletionRelevancePostfixMatch::Exact)
415            } else {
416                cov_mark::hit!(postfix_inexact_match_is_low_priority);
417                Some(CompletionRelevancePostfixMatch::NonExact)
418            };
419            let relevance = CompletionRelevance { postfix_match, ..Default::default() };
420            item.set_relevance(relevance);
421            item
422        }
423    }
424    Some(build(ctx, cap, delete_range))
425}

427fn add_custom_postfix_completions(
428    acc: &mut Completions,
429    ctx: &CompletionContext<'_>,
430    postfix_snippet: impl Fn(&str, &str, &str) -> Builder,
431    receiver_text: &str,
432) -> Option<()> {
433    ImportScope::find_insert_use_container(&ctx.token.parent()?, &ctx.sema)?;
434    ctx.config.postfix_snippets().filter(|(_, snip)| snip.scope == SnippetScope::Expr).for_each(
435        |(trigger, snippet)| {
436            let imports = match snippet.imports(ctx) {
437                Some(imports) => imports,
438                None => return,
439            };
440            let body = snippet.postfix_snippet(receiver_text);
441            let mut builder =
442                postfix_snippet(trigger, snippet.description.as_deref().unwrap_or_default(), &body);
443            builder.documentation(Documentation::new(format!("```rust\n{body}\n```")));
444            for import in imports.into_iter() {
445                builder.add_import(import);
446            }
447            builder.add_to(acc, ctx.db);
448        },
449    );
450    None
451}

453pub(crate) fn is_in_condition(it: &ast::Expr) -> bool {
454    it.syntax()
455        .parent()
456        .and_then(|parent| {
457            Some(match_ast! { match parent {
458                ast::IfExpr(expr) => expr.condition()? == *it,
459                ast::WhileExpr(expr) => expr.condition()? == *it,
460                ast::MatchGuard(guard) => guard.condition()? == *it,
461                ast::BinExpr(bin_expr) => (bin_expr.op_token()?.kind() == T![&&])
462                    .then(|| is_in_condition(&bin_expr.into()))?,
463                ast::Expr(expr) => (expr.syntax().text_range().start() == it.syntax().text_range().start())
464                    .then(|| is_in_condition(&expr))?,
465                _ => return None,
466            } })
467        })
468        .unwrap_or(false)
469}

807    fn postfix_completion_for_block(kind: &str) {
808        check_edit(kind, r#"fn main() { foo.$0 }"#, &format!("fn main() {{ {kind} {{ foo }} }}"));
809        check_edit(
810            kind,
811            r#"fn main() { { foo }.$0 }"#,
812            &format!("fn main() {{ {kind} {{ foo }} }}"),
813        );
814        check_edit(
815            kind,
816            r#"fn main() { if x { foo }.$0 }"#,
817            &format!("fn main() {{ {kind} {{ if x {{ foo }} }} }}"),
818        );
819        check_edit(
820            kind,
821            r#"fn main() { loop { foo }.$0 }"#,
822            &format!("fn main() {{ {kind} {{ loop {{ foo }} }} }}"),
823        );
824        check_edit(
825            kind,
826            r#"fn main() { if true {}.$0 }"#,
827            &format!("fn main() {{ {kind} {{ if true {{}} }} }}"),
828        );
829        check_edit(
830            kind,
831            r#"fn main() { while true {}.$0 }"#,
832            &format!("fn main() {{ {kind} {{ while true {{}} }} }}"),
833        );
834        check_edit(
835            kind,
836            r#"fn main() { for i in 0..10 {}.$0 }"#,
837            &format!("fn main() {{ {kind} {{ for i in 0..10 {{}} }} }}"),
838        );
839        check_edit(
840            kind,
841            r#"fn main() { let x = if true {1} else {2}.$0 }"#,
842            &format!("fn main() {{ let x = {kind} {{ if true {{1}} else {{2}} }} }}"),
843        );
844
845        // completion will not be triggered
846        check_edit(
847            kind,
848            r#"fn main() { let x = true else {panic!()}.$0}"#,
849            &format!("fn main() {{ let x = true else {{panic!()}}.{kind} $0}}"),
850        );
851    }
